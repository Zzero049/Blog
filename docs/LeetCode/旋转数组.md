class Solution {
    public void rotate(int[][] matrix) {
        int rowLen = matrix.length;
        int columnLen = matrix[0].length;
        int temp;
        if((rowLen==0||columnLen==0)&&rowLen!=columnLen) return;
        for(int i=0;i<rowLen;i++){
            for(int j=i;j<columnLen;j++){
                temp = matrix[j][i];
                matrix[j][i] = matrix[i][j];
                matrix[i][j] = temp;
            }
        }

        for(int i=0;i<columnLen;i++){
            for(int j=0;j<columnLen/2;j++){
                temp = matrix[i][j];
                matrix[i][j] = matrix[i][columnLen-j-1];
                matrix[i][columnLen-j-1] = temp;
            }
        }
    }
}
