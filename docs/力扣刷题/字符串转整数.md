class Solution {
    public int myAtoi(String str) {
        String sb = str.trim();
        int n = sb.length();
        int limit,ans=0;
        long temp = 0;

        if(n==0){
            return 0;
        }
        if(judgeFirstChar(sb)==1){
            limit = Integer.MAX_VALUE;
            String s = sb.substring(1);

            char [] chars = s.toCharArray();
            for(char c:chars){
                if(c>='0'&&c<='9'){
                    temp = temp*10 + c - '0';
                    if(temp>limit){
                        return limit;
                    }
                }else {
                    break;
                }
            }
            ans = (int)temp;
            return ans;
        }else if(judgeFirstChar(sb)==2){
            limit = Integer.MIN_VALUE;
            String s = sb.substring(1);
            int i=1;
            char [] chars = s.toCharArray();
            for(char c:chars){
                if(c>='0'&&c<='9'){
                    temp = temp*10 + c - '0';
                    i *= 10;
                    if(-temp<limit){
                        return limit;
                    }
                }else {
                    break;
                }
            }
            ans = (int) - temp;
            return ans;
        }else if(judgeFirstChar(sb)==3){
            limit = Integer.MAX_VALUE;
            char [] chars = sb.toCharArray();
            for(char c:chars){
                if(c>='0'&&c<='9'){
                    temp = temp*10 + c - '0';
                    if(temp>limit){
                        return limit;
                    }
                }else {
                    break;
                }
            }
            ans = (int)temp;
            return ans;
        }else{
            return 0;
        }
        }
    public int judgeFirstChar(String str){
        if(str.charAt(0)=='+'){
            return 1;
        }else if(str.charAt(0)=='-'){
            return 2;
        }else if(str.charAt(0)>='0'&&str.charAt(0)<='9'){
            return 3;
        }else {
            return 0;
        }
    }
}
