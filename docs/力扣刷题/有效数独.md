index = (i/3)*3 + j/3 (行优先)
```java
class Solution {
    public boolean isValidSudoku(char[][] board) {
        int row = board.length;
        int column;
        Map<Integer,Set<Character>> maps = new HashMap<>();
        for(int i=0;i<9;i++){
            Set<Character> tempSet = new HashSet<>();
            maps.put(i,tempSet);
        }
        Map<Integer,Set<Character>> mapsNine = new HashMap<>();
        for(int i=2;i<9;i+=3){
            for(int j=2;j<9;j+=3){
                Set<Character> tempSet = new HashSet<>();
                mapsNine.put(10*(i/3) +j/3,tempSet);
            }
        }

        for(int i=0;i<row;i++){
            column = board[i].length;
            Set<Character> tempSetRow = new HashSet<>();
            if(column!=9||row!=9){
                return false;
            }
            for(int j=0;j<column;j++){
                if(board[i][j] != '.'){
                    if(tempSetRow.contains(board[i][j])) return false;
                    tempSetRow.add(board[i][j]);
                    if(maps.get(j).contains(board[i][j])) return false;
                    maps.get(j).add(board[i][j]);
                    if(mapsNine.get(10*(i/3)+j/3).contains(board[i][j])) return false;
                    mapsNine.get(10*(i/3)+j/3).add(board[i][j]);
                }
            }
        }
        return true;
    }
}
```

用数组更方便，把列下标作为数独值进行判断，占用空间更小且更快
```java
class Solution {
    //官方用的hashmap，但采用数组更方便
    public boolean isValidSudoku(char[][] board) {
        int[][] rows = new int[9][9];
        int[][] col = new int[9][9];
        int[][] sbox = new int[9][9];
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                if (board[i][j]!='.'){
                    int num = board[i][j] - '1';
                    int index_box = (i/3)*3+j/3;
                    if (rows[i][num]==1) { return false;}
                    else { rows[i][num]=1; }
                    if (col[j][num]==1) { return false;}
                    else { col[j][num]=1; }
                    if (sbox[index_box][num]==1)  { return false;}
                    else { sbox[index_box][num]=1; }
                }
            }
        }
        return true;
    }
}
```
